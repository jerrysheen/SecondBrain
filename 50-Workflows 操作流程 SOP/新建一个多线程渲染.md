### 一次性任务线程：
````cpp
std::mutex m;
std::condition_variable cv;
bool start_flag = false;

void one_time_task()
{
	std::unique_lock<std::mutex> lk(m);
	cv.wait(lk,[]{return start_flag});
	lk.unlock();
	// do real work
}

int mian()
{
	std::tread th(one_time_task);
	// later
	{
		std::safe_guard<std::mutex> lock(m);
		start_flag = true;
	}
	cv.notify_one();
	th.join();
}
````
